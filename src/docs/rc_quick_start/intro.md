const copyText = async (text, event) => {
  try {
    await navigator.clipboard.writeText(text);
    const btn = event.currentTarget;
    btn.classList.add('copied');
    setTimeout(() => btn.classList.remove('copied'), 1500);
  } catch (err) {
    console.error('复制失败:', err);
  }
};
